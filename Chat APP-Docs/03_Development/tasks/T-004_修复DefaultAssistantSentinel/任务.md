## 方案 B 内容

修改 `src/renderer/src/services/AssistantService.ts:173-178` 的 `mapLegacyTopicToDto`：

```ts
// before
export function mapLegacyTopicToDto(topic: Topic): CreateTopicDto {
  return {
    name: topic.name,
    assistantId: topic.assistantId
  }
}

// after
export function mapLegacyTopicToDto(topic: Topic): CreateTopicDto {
  return {
    name: topic.name,
    assistantId: topic.assistantId === 'default' ? null : topic.assistantId
  }
}
```
